'#0f0' : '#0a0'; ctx.fillRect(seg.x * TILE, seg.y * TILE, TILE - 1, TILE - 1); }); // Food ctx.fillStyle = '#f00'; ctx.fillRect(food.x * TILE, food.y * TILE, TILE - 1, TILE - 1); } function update() { if (!started || gameOver) return; const head = { x: snake[0].x + direction.x, y: snake[0].y + direction.y }; // Wall collision if (head.x < 0 || head.x >= GRID || head.y < 0 || head.y >= GRID) { gameOver = true; messageEl.textContent = `Game Over! Press any arrow key to restart`; return; } // Self collision if (snake.some(seg => seg.x === head.x && seg.y === head.y)) { gameOver = true; messageEl.textContent = `Game Over! Press any arrow key to restart`; return; } snake.unshift(head); // Eat food if (head.x === food.x && head.y === food.y) { score += 10; scoreEl.textContent = `Score: ${score}`; placeFood(); } else { snake.pop(); } draw(); } document.addEventListener('keydown', (e) => { const key = { ArrowUp: { x: 0, y: -1 }, ArrowDown: { x: 0, y: 1 }, ArrowLeft: { x: -1, y: 0 }, ArrowRight: { x: 1, y: 0 } }; if (!key[e.key]) return; e.preventDefault(); if (gameOver) { reset(); direction = key[e.key]; started = true; return; } if (!started) { started = true; direction = key[e.key]; messageEl.textContent = ''; } // Prevent reverse if ((direction.x + key[e.key].x !== 0) || (direction.y + key[e.key].y !== 0)) { direction = key[e.key]; } }); reset(); setInterval(update, 1000 / 60); ``` ## Quick Start Save all three files in the same folder, then open `index.html` in a browser. Click the page and use **arrow keys** to control the snake. The game is fully self-contained — no build step needed.
